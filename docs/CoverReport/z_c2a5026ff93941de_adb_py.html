50 def run_cmd(self, extra_args, disable_log=False): 

51 """ 

52 run an adb command and return the output 

53 :return: output of adb command 

54 @param extra_args: arguments to run in adb 

55 """ 

56 try: 

57 

58 if isinstance(extra_args, str) or isinstance(extra_args, str): 

59 extra_args = extra_args.split() 

60 if not isinstance(extra_args, list): 

61 msg = "invalid arguments: %s\nshould be list or str, %s given" % (extra_args, type(extra_args)) 

62 self.logger.warning(msg) 

63 raise ADBException(msg) 

64 

65 args = [] + self.cmd_prefix 

66 args += extra_args 

67 

68 self.logger.debug('command:') 

69 self.logger.debug(args) 

70 if not disable_log: 

71 r = subprocess.check_output(args).strip() 

72 else: 

73 # redirect the output to stderr when disable ouput 

74 # which will avoid logging in the terminal 

75 r = subprocess.check_output( 

76 args, 

77 stderr=subprocess.STDOUT 

78 ) 

79 if not isinstance(r, str): 

80 r = r.decode() 

81 self.logger.debug('return:') 

82 self.logger.debug(r) 

83 return r 

84 except subprocess.CalledProcessError as e: 

85 self.logger.warning('error in running command: %s' % e)

166 def get_display_info(self): 

167 """ 

168 Gets C{mDefaultViewport} and then C{deviceWidth} and C{deviceHeight} values from dumpsys. 

169 This is a method to obtain display dimensions and density 

170 """ 

171 display_info = {} 

172 logical_display_re = re.compile(".*DisplayViewport{valid=true, .*orientation=(?P<orientation>\d+)," 

173 " .*deviceWidth=(?P<width>\d+), deviceHeight=(?P<height>\d+).*") 

174 dumpsys_display_result = self.shell("dumpsys display") 

175 if dumpsys_display_result is not None: 

176 for line in dumpsys_display_result.splitlines(): 

177 m = logical_display_re.search(line, 0) 

178 if m: 

179 for prop in ['width', 'height', 'orientation']: 

180 display_info[prop] = int(m.group(prop)) 

181 

182 if 'width' not in display_info or 'height' not in display_info: 

183 physical_display_re = re.compile('Physical size: (?P<width>\d+)x(?P<height>\d+)') 

184 m = physical_display_re.search(self.shell('wm size')) 

185 if m: 

186 for prop in ['width', 'height']: 

187 display_info[prop] = int(m.group(prop)) 

188 

189 if 'width' not in display_info or 'height' not in display_info: 

190 # This could also be mSystem or mOverscanScreen 

191 display_re = re.compile('\s*mUnrestrictedScreen=\((?P<x>\d+),(?P<y>\d+)\) (?P<width>\d+)x(?P<height>\d+)') 

192 # This is known to work on older versions (i.e. API 10) where mrestrictedScreen is not available 

193 display_width_height_re = re.compile('\s*DisplayWidth=(?P<width>\d+) *DisplayHeight=(?P<height>\d+)') 

194 for line in self.shell('dumpsys window').splitlines(): 

195 m = display_re.search(line, 0) 

196 if not m: 

197 m = display_width_height_re.search(line, 0) 

198 if m: 

199 for prop in ['width', 'height']: 

200 display_info[prop] = int(m.group(prop)) 

201 

202 if 'orientation' not in display_info: 

203 surface_orientation_re = re.compile("SurfaceOrientation:\s+(\d+)") 

204 output = self.shell("dumpsys input") 

205 m = surface_orientation_re.search(output) 

206 if m: 

207 display_info['orientation'] = int(m.group(1)) 

208 

209 density = None 

210 float_re = re.compile(r"[-+]?\d*\.\d+|\d+") 

211 d = self.get_property('ro.sf.lcd_density') 

212 if float_re.match(d): 

213 density = float(d) 

214 else: 

215 d = self.get_property('qemu.sf.lcd_density') 

216 if float_re.match(d): 

217 density = float(d) 

218 else: 

219 physical_density_re = re.compile('Physical density: (?P<density>[\d.]+)', re.MULTILINE) 

220 m = physical_density_re.search(self.shell('wm density')) 

221 if m: 

222 density = float(m.group('density')) 

223 if density is not None: 

224 display_info['density'] = density 

225 

226 display_info_keys = {'width', 'height', 'orientation', 'density'} 

227 if not display_info_keys.issuperset(display_info): 

228 self.logger.warning("getDisplayInfo failed to get: %s" % display_info_keys) 

229 

230 return display_info

302 def __transform_point_by_orientation(self, xy, orientation_orig, orientation_dest): 

303 (x, y) = xy 

304 if orientation_orig != orientation_dest: 

305 if orientation_dest == 1: 

306 _x = x 

307 x = self.get_display_info()['width'] - y 

308 y = _x 

309 elif orientation_dest == 3: 

310 _x = x 

311 x = y 

312 y = self.get_display_info()['height'] - _x 

313 return x, y

347 def drag(self, start_xy, end_xy, duration, orientation=-1): 

348 """ 

349 Sends drag event n PX (actually it's using C{input swipe} command. 

350 @param start_xy: starting point in pixel 

351 @param end_xy: ending point in pixel 

352 @param duration: duration of the event in ms 

353 @param orientation: the orientation (-1: undefined) 

354 """ 

355 (x0, y0) = start_xy 

356 (x1, y1) = end_xy 

357 if orientation == -1: 

358 orientation = self.get_orientation() 

359 (x0, y0) = self.__transform_point_by_orientation((x0, y0), orientation, self.get_orientation()) 

360 (x1, y1) = self.__transform_point_by_orientation((x1, y1), orientation, self.get_orientation()) 

361 

362 version = self.get_sdk_version() 

363 if version <= 15: 

364 self.logger.error("drag: API <= 15 not supported (version=%d)" % version) 

365 elif version <= 17: 

366 self.shell("input swipe %d %d %d %d" % (x0, y0, x1, y1)) 

367 else: 

368 self.shell("input touchscreen swipe %d %d %d %d %d" % (x0, y0, x1, y1, duration))

397 # clear the app data (including user data, cache) 

398 def clear_app_data(self, package_name: str): 

399 """ 

400 clear the app data (including user data and cache) 

401 :param package_name: the app package name 

402 :return: 

403 """ 

404 # By default, "pm clear" will clear an app's private data and public data. 

405 # private data: /data/data/#{package_name}/ 

406 # public data: /sdcard/Android/data/#{package_name}/ 

407 out = self.shell("pm clear %s" % package_name) 

408 logging.info("clear app data: %s" % out) 

409 

410 # Some apps may dump data to external storage (e.g., SDCard), so we need to add additional data clear commands 

411 if package_name.startswith('nl.mpcjanssen.simpletask'): 

412 self.shell('rm -rf /sdcard/data/nl.mpcjanssen.simpletask') 

413 logging.info('clear data on external storage for nl.mpcjanssen.simpletask') 

414 elif package_name.startswith('net.gsantner.markor'): 

415 self.shell('rm -rf /sdcard/Documents/markor') 

416 logging.info('clear data on external storage for net.gsantner.markor') 

417 elif package_name.startswith('net.programmierecke.radiodroid2'): 

418 self.shell('rm -rf /sdcard/Music/Recordings') 

419 logging.info('clear data on external storage for net.programmierecke.radiodroid2') 

420 elif package_name.startswith('free.rm.skytube.extra'): 

421 self.shell('rm -f /sdcard/Movies/*.mp4') 

422 logging.info('clear data on external storage for free.rm.skytube.extra') 

423 elif package_name.startswith('com.ichi2.anki'): 

424 self.shell('rm -rf /sdcard/AnkiDroid') 

425 logging.info('clear data on external storage for free.rm.skytube.extra') 

426 elif package_name.startswith('org.liberty.android.fantastischmemo'): 

427 self.shell('rm -rf /sdcard/anymemo') 

428 logging.info('clear data on external storage for org.liberty.android.fantastischmemo')
